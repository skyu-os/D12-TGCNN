# 接入交通预测的动态路径规划算法开发文档

## 1. 系统概述

### 1.1 背景
本项目旨在实现基于深度学习交通预测（TGCN）的动态路径规划系统，通过预测未来时段的路网交通状态，为用户提供更加智能和准确的路径规划服务。

### 1.2 核心目标
- **实时性**：支持实时接入交通预测数据，动态调整路径规划结果
- **准确性**：基于预测的交通速度和流量，计算更准确的行驶时间
- **智能化**：结合多目标优化和路口约束，提供个性化路径建议
- **可扩展性**：支持不同预测模型和数据源的灵活接入

### 1.3 技术栈
- **交通预测**: TGCN (Temporal Graph Convolutional Network)
- **数据源**: PEMS D12 交通传感器数据
- **路径规划**: 增强版A*算法 + 多目标优化
- **传感器插值**: APN自适应插值方法
- **路网构建**: OpenStreetMap + PEMS D12 传感器映射

## 4. 动态边权建模

### 4.1 时变边权计算

```python
class DynamicEdgeWeightManager:
    """
    动态边权管理器
    """
    def __init__(
        self,
        road_graph,
        traffic_predictor: TGCNPredictor,
        sensor_mapper: SensorRoadMapper
    ):
        """
        Args:
            road_graph: 路网图对象
            traffic_predictor: 交通预测器
            sensor_mapper: 传感器-道路映射器
        """
        self.road_graph = road_graph
        self.predictor = traffic_predictor
        self.mapper = sensor_mapper
        
    def calculate_edge_weight(
        self,
        u: int,
        v: int,
        key: int,
        departure_time: int,  # 出发时间（Unix时间戳）
        vehicle_type: str = 'gasoline'
    ) -> float:
        """
        计算边的动态权重（行驶时间）
        
        Args:
            u: 起始节点
            v: 目标节点
            key: 边键
            departure_time: 预计到达该边的时间
            vehicle_type: 车辆类型
            
        Returns:
            边的权重（秒）
        """
        # 1. 获取边的基础信息
        length = self.road_graph.get_edge_length(u, v, key)
        road_type = self.road_graph.get_edge_type(u, v, key)
        
        # 2. 获取该边对应的传感器
        sensors = self.mapper.get_sensors_for_edge(u, v, key)
        
        if not sensors:
            # 无传感器数据，使用默认速度
            default_speed = self._get_default_speed(road_type)
            return length / (default_speed / 3.6)
        
        # 3. 获取预测速度
        predicted_speed = self._get_predicted_speed(
            sensors, departure_time, vehicle_type
        )
        
        # 4. 计算行驶时间
        travel_time = length / (predicted_speed / 3.6)
        
        return travel_time
        
    def _get_predicted_speed(
        self,
        sensors: List[int],
        time: int,
        vehicle_type: str
    ) -> float:
        """
        获取预测速度
        
        Args:
            sensors: 相关传感器列表
            time: 预测时间
            vehicle_type: 车辆类型
            
        Returns:
            预测速度 (km/h)
        """
        speeds = []
        for sensor_id in sensors:
            speed = self.predictor.get_sensor_speed(
                sensor_id, 
                future_minutes=0  # 需要计算时间差
            )
            if speed > 0:
                speeds.append(speed)
        
        if not speeds:
            return self._get_default_speed('primary')
        
        # 使用加权平均或最小值
        return min(speeds)  # 保守策略
        
    def _get_default_speed(self, road_type: str) -> float:
        """获取道路类型的默认速度"""
        default_speeds = {
            'motorway': 100.0,
            'trunk': 80.0,
            'primary': 60.0,
            'secondary': 50.0,
            'tertiary': 40.0,
            'residential': 30.0,
        }
        return default_speeds.get(road_type, 40.0)
```

class TimeDependentAStar:
    """
    时变A*路径规划算法
    """
    def __init__(
        self,
        road_graph,
        weight_manager: DynamicEdgeWeightManager,
        constraints: IntersectionConstraints
    ):
        """
        Args:
            road_graph: 路网图
            weight_manager: 动态边权管理器
            constraints: 路口约束管理器
        """
        self.G = road_graph.G
        self.weight_manager = weight_manager
        self.constraints = constraints
        
    def find_path(
        self,
        start_node: int,
        end_node: int,
        departure_time: int,
        vehicle_type: str = 'gasoline',
        objective: str = 'time'
    ) -> Optional[Dict]:
        """
        时变A*搜索
        
        Args:
            start_node: 起点节点ID
            end_node: 终点节点ID
            departure_time: 出发时间（Unix时间戳）
            vehicle_type: 车辆类型
            objective: 优化目标
            
        Returns:
            路径信息字典
        """
        import heapq
        
        # 优先队列: (f_score, current_time, node, prev_node)
        open_set = [(0, departure_time, start_node, None)]
        
        came_from = {}
        g_score = {start_node: departure_time}  # 存储到达时间
        
        while open_set:
            f_score, current_time, current, prev_node = heapq.heappop(open_set)
            
            if current == end_node:
                return self._reconstruct_path(
                    came_from, current, current_time, departure_time
                )
            
            # 遍历邻居
            for neighbor in self.G.neighbors(current):
                for key in self.G[current][neighbor]:
                    # 计算动态边权（考虑当前时间）
                    edge_time = self.weight_manager.calculate_edge_weight(
                        current, neighbor, key, 
                        departure_time=current_time,
                        vehicle_type=vehicle_type
                    )
                    
                    # 计算路口约束代价
                    constraint_time = self._calculate_constraint_time(
                        current, neighbor, prev_node, current_time
                    )
                    
                    # 到达邻居节点的时间
                    arrival_time = current_time + edge_time + constraint_time
                    
                    # 更新g_score
                    if arrival_time < g_score.get(neighbor, float('inf')):
                        came_from[neighbor] = (current, key, current_time, arrival_time)
                        g_score[neighbor] = arrival_time
                        
                        # 计算启发式函数
                        h_score = self._heuristic(neighbor, end_node)
                        f_score = arrival_time + h_score
                        
                        heapq.heappush(open_set, (
                            f_score, arrival_time, neighbor, current
                        ))
        
        return None  # 不可达
