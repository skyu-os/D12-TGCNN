基于时空图神经网络的交通状态预测与路径规划系统

- 时间范围：2026-01-01 至 2026-02-28，共59天
- 路网来源：OpenStreetMap（WGS-84坐标系）

```python
# predictor.py

import torch
import numpy as np

class TrafficPredictor:
    def __init__(self, model_path, model_type="lstm"):
        self.model_type = model_type
        self.model = self._load_model(model_path)
        self.model.eval()

    def _load_model(self, path):
        # 加载已训练好的模型
        model = torch.load(path, map_location='cpu')
        return model

    def predict(self, history_speed, steps=3):
        """
        history_speed: np.array, shape [seq_len, num_sensors]
        返回: np.array, shape [steps, num_sensors]
        """
        with torch.no_grad():
            x = torch.FloatTensor(history_speed).unsqueeze(0)
            pred = self.model(x)
            return pred.squeeze(0).numpy()[:steps]

    def predict_single_sensor(self, sensor_id, history_speed, steps=3):
        """预测单个传感器"""
        pred = self.predict(history_speed, steps)
        return pred[:, sensor_id]

## **Module 3：路网权重更新模块**

### **职责**
- 把预测速度转换为路段通行时间
- 生成 OSRM 可用的自定义权重文件
- 或者直接用权重做 Dijkstra 自定义路径计算

### **核心逻辑**

把传感器预测速度映射到路段：

$$
t_{edge} = \frac{L_{edge}}{v_{sensor}} \times 3600
$$

其中：
- \(t_{edge}\) 为路段通行时间（秒）
- \(L_{edge}\) 为路段长度（km）
- \(v_{sensor}\) 为对应传感器预测速度（km/h）

### **实现**

```python
# weight_updater.py

import pandas as pd
import numpy as np
import json

class WeightUpdater:
    def __init__(self, edges_path, sensor_match_path):
        self.edges = pd.read_csv(edges_path)
        self.sensor_match = pd.read_csv(sensor_match_path)

    def speed_to_travel_time(self, speed_kmh, length_km):
        """速度转通行时间（秒）"""
        if speed_kmh <= 0:
            return 9999
        return (length_km / speed_kmh) * 3600

    def update_weights(self, predicted_speeds):
        """
        predicted_speeds: np.array, shape [1062]
        返回: dict {edge_id: travel_time_seconds}
        """
        weights = {}
        for _, edge in self.edges.iterrows():
            edge_id = edge['edge_id']
            sensor_id = self._get_nearest_sensor(edge)

            if sensor_id is not None:
                speed = predicted_speeds[sensor_id]
                # 反归一化
                speed_kmh = speed * 120  # 假设最大速度120km/h
            else:
                speed_kmh = 60  # 默认速度

            length_km = edge['length'] / 1000
            travel_time = self.speed_to_travel_time(speed_kmh, length_km)
            weights[edge_id] = travel_time

        return weights

    def _get_nearest_sensor(self, edge):
        """找路段对应的最近传感器"""
        matches = self.sensor_match[
            self.sensor_match['osm_node_id'] == edge['u']
        ]
        if len(matches) > 0:
            return matches.iloc[0]['sensor_idx']
        return None

# osrm_client.py

import requests

class OSRMClient:
    def __init__(self, host="http://localhost:5000"):
        self.host = host

    def route(self, start_lng, start_lat, end_lng, end_lat):
        """
        标准路径规划
        返回: dict with distance, duration, geometry
        """
        url = (
            f"{self.host}/route/v1/driving/"
            f"{start_lng},{start_lat};"
            f"{end_lng},{end_lat}"
            f"?overview=full&geometries=geojson&steps=true"
        )
        r = requests.get(url, timeout=5)
        r.raise_for_status()
        data = r.json()

        if data['code'] != 'Ok':
            raise ValueError(f"OSRM error: {data['code']}")

        route = data['routes'][0]
        return {
            'distance_m': route['distance'],
            'duration_s': route['duration'],
            'geometry': route['geometry'],
            'steps': route['legs'][0]['steps']
        }

    def health_check(self):
        """检查 OSRM 服务是否正常"""
        try:
            url = f"{self.host}/route/v1/driving/-117.8,33.7;-117.9,33.8"
            r = requests.get(url, timeout=3)
            return r.status_code == 200
        except:
            return False

predictor = TrafficPredictor("./models/lstm_best.pt")
sensors_df = pd.read_csv("./data/sensors_matched.csv")

# ==================== 健康检查 ====================
@app.route('/api/health')
def health():
    return jsonify({
        'status': 'ok',
        'osrm': osrm.health_check(),
        'model': 'lstm',
        'sensors': len(sensors_df)
    })

# ==================== 路径规划 ====================
@app.route('/api/route')
def route():
    try:
        start = request.args.get('start')  # "lng,lat"
        end = request.args.get('end')      # "lng,lat"
        mode = request.args.get('mode', 'standard')

        slng, slat = map(float, start.split(','))
        elng, elat = map(float, end.split(','))

        result = osrm.route(slng, slat, elng, elat)

        return jsonify({
            'code': 'Ok',
            'distance_km': round(result['distance_m'] / 1000, 2),
            'duration_min': round(result['duration_s'] / 60, 1),
            'geometry': result['geometry'],
            'steps': result['steps']
        })

    except Exception as e:
        return jsonify({'code': 'Error', 'message': str(e)}), 400

# ==================== 交通预测 ====================
@app.route('/api/predict')
def predict():
    try:
        sensor_id = int(request.args.get('sensor_id', 0))
        steps = int(request.args.get('steps', 3))

        # 用最近一段历史数据做预测（这里用随机数模拟，实际要读数据库）
        history = np.random.rand(12, 1062).astype(np.float32)
        pred = predictor.predict_single_sensor(sensor_id, history, steps)

        return jsonify({
            'code': 'Ok',
            'sensor_id': sensor_id,
            'predictions': pred.tolist(),
            'steps': steps,
            'interval_minutes': 5
        })

    except Exception as e:
        return jsonify({'code': 'Error', 'message': str(e)}), 400

# ==================== 传感器列表 ====================
@app.route('/api/sensors')
def sensors():
    data = sensors_df[['sensor_id', 'latitude', 'longitude']].to_dict('records')
    return jsonify({'code': 'Ok', 'count': len(data), 'sensors': data})

@app.route('/api/sensors/<int:sid>')
def sensor_detail(sid):
    row = sensors_df[sensors_df['sensor_id'] == sid]
    if len(row) == 0:
        return jsonify({'code': 'Error', 'message': 'Not found'}), 404
    return jsonify({'code': 'Ok', 'sensor': row.iloc[0].to_dict()})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8000, debug=True)

```plaintextjavascript
// main.js

const map = L.map('map').setView([33.75, -117.87], 11);

// OSM底图
L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '© OpenStreetMap contributors'
}).addTo(map);

let startMarker = null, endMarker = null, routeLine = null;
let clickCount = 0;

// 点击地图选点
map.on('click', async function(e) {
    const { lat, lng } = e.latlng;
    clickCount++;

    if (clickCount === 1) {
        if (startMarker) map.removeLayer(startMarker);
        startMarker = L.marker([lat, lng], {
            icon: greenIcon
        }).bindPopup('起点').addTo(map);
        document.getElementById('start-coord').textContent =
            `${lng.toFixed(4)}, ${lat.toFixed(4)}`;
    }

    if (clickCount === 2) {
        if (endMarker) map.removeLayer(endMarker);
        endMarker = L.marker([lat, lng], {
            icon: redIcon
        }).bindPopup('终点').addTo(map);
        document.getElementById('end-coord').textContent =
            `${lng.toFixed(4)}, ${lat.toFixed(4)}`;

        // 请求路径
        await planRoute();
        clickCount = 0;
    }
});

async function planRoute() {
    const startCoord = getMarkerCoord(startMarker);
    const endCoord = getMarkerCoord(endMarker);

    showLoading(true);

    try {
        const res = await fetch(
            `/api/route?start=${startCoord}&end=${endCoord}`
        );
        const data = await res.json();

        if (data.code !== 'Ok') throw new Error(data.message);

        // 画路线
        if (routeLine) map.removeLayer(routeLine);
        const coords = data.geometry.coordinates.map(c => [c[1], c[0]]);
        routeLine = L.polyline(coords, {
            color: '#3388ff',
            weight: 6,
            opacity: 0.8
        }).addTo(map);
        map.fitBounds(routeLine.getBounds());

        // 显示结果
        document.getElementById('distance').textContent =
            `${data.distance_km} km`;
        document.getElementById('duration').textContent =
            `${data.duration_min} 分钟`;

    } catch (err) {
        alert('路径规划失败: ' + err.message);
    } finally {
        showLoading(false);
    }
}
```
